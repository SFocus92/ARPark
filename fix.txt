Проблема найдена! MindAR ожидает **каждый `.mind` файл отдельно**, а ты передаёшь их одной строкой через запятую.

## Ошибка:
```
GET /assets/nft/marker-1,marker-2,marker-3,...,marker-7.mind 404
```
MindAR пытается загрузить **один файл** с именем `marker-1,marker-2,...,marker-7.mind`.

## Решение:

MindAR A-Frame компонент принимает `imageTargetSrc` как **URL к одному `.mind` файлу**. Для нескольких маркеров нужно использовать другой подход — либо:

Точно! Квест с 7 этапами, и каждый маркер должен быть **отдельным фото реального объекта в парке** — скамейка, дерево, статуя и т.д. Поэтому один файл `targets.mind` не подойдёт — все маркеры смешаются.

Нужно **загрузить 7 отдельных `.mind` файлов**. MindAR A-Frame компонент не поддерживает массив, но можно загружать через **JS API напрямую**.

## Правильное решение: MindAR JS API + отдельные маркеры

Вот исправленный `ar-scene.tsx`, который:
- Загружает MindAR **без A-Frame компонента**
- Загружает **каждый `.mind` файл отдельно** через `addImageTargets`
- Обрабатывает `targetIndex` для каждого маркера

```tsx
'use client';

import { useEffect, useRef, useState } from 'react';
import { useQuest } from '@/hooks/use-quest';
import { STEPS, QuestStep } from '@/lib/quest-config';

interface ARSceneProps {
  onReady?: () => void;
  onError?: (error: string) => void;
}

export function ARScene({ onReady, onError }: ARSceneProps) {
  const containerRef = useRef<HTMLDivElement>(null);
  const [isLoading, setIsLoading] = useState(true);
  const [status, setStatus] = useState('Инициализация...');
  const [error, setError] = useState<string | null>(null);
  const initRef = useRef(false);
  const scannerRef = useRef<any>(null);

  const {
    handleMarkerFound,
    setCameraReady,
    setCameraError,
    currentMarker,
    showingContent,
    completedSteps,
  } = useQuest();

  useEffect(() => {
    if (initRef.current) return;
    initRef.current = true;

    async function init() {
      try {
        // 1. Ждём A-Frame (даёт нам THREE)
        setStatus('Ожидание A-Frame...');
        await waitForGlobal('AFRAME', 15000);
        await waitForGlobal('THREE', 5000);
        console.log('[AR] A-Frame + THREE готовы');

        // 2. Загружаем MindAR (НЕ A-Frame версию, а THREE версию)
        setStatus('Загрузка MindAR...');
        
        // Используем динамический импорт ESM модуля
        const mindarModule = await import(
          'https://cdn.jsdelivr.net/npm/mind-ar@1.2.5/dist/mindar-image-three.prod.js'
        );

        // Ищем конструктор
        const MindARImage = 
          mindarModule.MindARImage || 
          mindarModule.default?.MindARImage ||
          (mindarModule as any).default;
        
        if (!MindARImage) {
          throw new Error('MindARImage не найден в модуле');
        }

        console.log('[AR] MindAR модуль загружен');

        // 3. Создаём сканер с ПЕРВЫМ маркером
        const firstMarker = STEPS.find(s => s.markerType === 'nft')!;
        
        setStatus('Создание сканера...');
        const scanner = new MindARImage({
          imageTargetSrc: firstMarker.nftDescriptor! + '.mind',
          filterMinCF: 0.1,
          filterBeta: 0.001,
        });

        // 4. Добавляем ОСТАЛЬНЫЕ маркеры
        const remainingMarkers = STEPS
          .filter(s => s.markerType === 'nft' && s.id !== firstMarker.id)
          .map(s => s.nftDescriptor! + '.mind');

        if (remainingMarkers.length > 0) {
          setStatus(`Загрузка ${remainingMarkers.length} маркеров...`);
          
          // Загружаем каждый .mind файл и добавляем в сканер
          for (const markerPath of remainingMarkers) {
            const response = await fetch(markerPath);
            const buffer = await response.arrayBuffer();
            scanner.addImageTargets(new Uint8Array(buffer));
            console.log('[AR] Добавлен маркер:', markerPath);
          }
        }

        // 5. Обработчики
        scanner.onTargetFound = (targetIndex: number) => {
          console.log(`[AR] 🎯 Маркер найден! Индекс: ${targetIndex}`);
          const step = STEPS.filter(s => s.markerType === 'nft')[targetIndex];
          if (step) {
            handleMarkerFound(step.id);
            setStatus(`✅ ${step.title}`);
          }
        };

        scanner.onTargetLost = () => {
          setStatus('🔍 Ищите маркер...');
        };

        // 6. Запуск
        setStatus('Запуск камеры...');
        await scanner.start(containerRef.current!);
        
        scannerRef.current = scanner;
        console.log('[AR] ✅ Сканер запущен!');
        
        setIsLoading(false);
        setCameraReady(true);
        onReady?.();
        setStatus('📷 Наведите камеру на маркер');

      } catch (err: any) {
        console.error('[AR] ❌ Ошибка:', err);
        setError(err.message);
        setCameraError(err.message);
        onError?.(err.message);
        setIsLoading(false);
      }
    }

    init();

    return () => {
      if (scannerRef.current?.stop) {
        scannerRef.current.stop();
      }
    };
  }, []);

  const currentStep = STEPS[completedSteps];
  const displayStep = currentMarker && showingContent
    ? STEPS.find(s => s.id === currentMarker)
    : null;

  return (
    <div ref={containerRef} style={{ position: 'fixed', inset: 0, zIndex: 0, background: '#000' }}>
      <StatusBar status={status} />
      <ProgressBar current={completedSteps} total={STEPS.length} title={currentStep?.title} />
      {displayStep && <MarkerOverlay step={displayStep} />}
      {isLoading && <LoadingScreen status={status} />}
      {error && <ErrorScreen error={error} />}
    </div>
  );
}

function waitForGlobal(name: string, timeoutMs: number): Promise<void> {
  return new Promise((resolve, reject) => {
    const start = Date.now();
    const check = setInterval(() => {
      if ((window as any)[name]) {
        clearInterval(check);
        resolve();
      } else if (Date.now() - start > timeoutMs) {
        clearInterval(check);
        reject(new Error(`Timeout: ${name}`));
      }
    }, 100);
  });
}
```
